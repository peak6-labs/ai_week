# PnL UI: Realized vs Unrealized Breakdown + Closed Positions Fix

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix the always-empty closed positions table by switching from Supabase to a Kalshi-fills-based poller, and add realized P&L to the status bar and a totals row to the closed positions table.

**Architecture:** A new `_poll_fills()` async task fetches all fills from `GET /portfolio/fills` every 5 minutes (paginated). A pure `compute_closed_positions()` function reconstructs per-market closed positions from the fills cache. Results are written to `trading_state.closed_positions` which the existing UI already renders. The old Supabase `_poll_closed_positions` poller is removed.

**Tech Stack:** Python asyncio, Kalshi REST API v2 (`/portfolio/fills`), FastAPI, vanilla JS + HTML.

```python
def compute_closed_positions(
    fills_cache: dict[str, list[dict]],
    open_tickers: set[str],
) -> list[dict]:
    """Reconstruct closed positions from a per-ticker fills cache.

    A position is closed when total sell count >= total buy count for a ticker
    AND that ticker is not in the current open positions set. Returns rows
    sorted newest-closed-first, matching the shape expected by
    renderClosedPositions() in the UI.
    """
    result: list[dict] = []
    for ticker, fills in fills_cache.items():
        if ticker in open_tickers:
            continue
        buy_fills = [f for f in fills if f.get("action") == "buy"]
        sell_fills = [f for f in fills if f.get("action") == "sell"]
        total_bought = sum(f.get("count", 0) for f in buy_fills)
        total_sold = sum(f.get("count", 0) for f in sell_fills)
        if total_bought == 0 or total_sold < total_bought:
            continue

        side_raw = (buy_fills[0].get("side") or "yes").lower()
        side = "YES" if side_raw == "yes" else "NO"

        def _avco(fill_list: list[dict], use_no_price: bool) -> float:
            total_weight = sum(f.get("count", 0) for f in fill_list)
            if total_weight == 0:
                return 0.0
            weighted_sum = sum(
                f.get("count", 0) * (
                    (100 - f.get("yes_price", 0)) if use_no_price
                    else f.get("yes_price", 0)
                )
                for f in fill_list
            )
            return weighted_sum / total_weight

        use_no = side == "NO"
        entry_price_cents = _avco(buy_fills, use_no)
        exit_price_cents = _avco(sell_fills, use_no)
        realized_pnl_dollars = (exit_price_cents - entry_price_cents) * total_bought / 100.0

        opened_at = min(f.get("created_time", "") for f in buy_fills)
        closed_at = max(f.get("created_time", "") for f in sell_fills)

        result.append({
            "ticker": ticker,
            "side": side,
            "contracts": total_bought,
            "entry_price_cents": entry_price_cents,
            "exit_price_cents": exit_price_cents,
            "opened_at": opened_at,
            "closed_at": closed_at,
            "realized_pnl_dollars": round(realized_pnl_dollars, 4),
        })

    result.sort(key=lambda row: row["closed_at"], reverse=True)
    return result
```

async def _poll_fills(trading_state: TradingState) -> None:
    """Rebuild closed positions from Kalshi fills every 5 minutes.

    Fetches all portfolio fills via paginated GET /portfolio/fills, groups them
    by ticker, and calls compute_closed_positions() to derive the closed
    positions list. Replaces the Supabase-based _poll_closed_positions poller.
    """
    from kalshi_trader.client import KalshiClient
    from kalshi_trader.dashboard.read_only_client import ReadOnlyKalshiClient

    await asyncio.sleep(4)
    trading_state.log("Fills poller started")

    async with KalshiClient() as raw_client:
        client = ReadOnlyKalshiClient(raw_client)
        while True:
            try:
                fills_cache: dict[str, list[dict]] = {}
                cursor: str | None = None

                while True:
                    response = await client.get_fills(cursor=cursor)
                    page_fills = response.get("fills") or []
                    for fill in page_fills:
                        ticker = fill.get("ticker", "")
                        if ticker:
                            fills_cache.setdefault(ticker, []).append(fill)
                    next_cursor = response.get("cursor")
                    if not next_cursor or not page_fills:
                        break
                    cursor = next_cursor

                open_tickers = {
                    p["ticker"] for p in trading_state.positions if p.get("ticker")
                }
                trading_state.closed_positions = compute_closed_positions(
                    fills_cache, open_tickers
                )

            except asyncio.CancelledError:
                raise
            except Exception as caught_exception:
                logger.warning("Fills poll failed: %s", caught_exception)

            await asyncio.sleep(300)  # 5 minutes
```

function renderClosedPositions(positions) {
  const tbody = document.getElementById('closed-positions-tbody');
  const tfoot = document.getElementById('closed-positions-tfoot');
  if (!positions || positions.length === 0) {
    tbody.innerHTML = '<tr class="empty-row"><td colspan="8">No closed positions</td></tr>';

    const cpUrl = p.ticker ? `https://kalshi.com/markets/${p.ticker.split('-')[0].toLowerCase()}` : null;
    const cpTickerEl = cpUrl

    const entryDisplay = p.entry_price_cents != null ? Math.round(p.entry_price_cents) + '¢' : '—';

    const pnl = p.realized_pnl_dollars;
    const pnlDisplay = pnl != null

      : '<span style="color:var(--text-tertiary)">—</span>';
    return `<tr>
      <td>${cpTickerEl}</td>
      <td>${sideBadge}</td>
      <td>${p.contracts ?? '—'}</td>
      <td style="color:var(--text-secondary)">${escHtml(timeDateTime(p.opened_at))}</td>
      <td style="color:var(--text-secondary)">${escHtml(timeDateTime(p.closed_at))}</td>
      <td>${escHtml(entryDisplay)}</td>
      <td>${escHtml(exitDisplay)}</td>
      <td>${pnlDisplay}</td>
    </tr>`;
  }).join('');

  // Totals row
  const totalContracts = positions.reduce((sum, p) => sum + (p.contracts ?? 0), 0);
  const totalPnl = positions.reduce((sum, p) => sum + (p.realized_pnl_dollars ?? 0), 0);
  document.getElementById('closed-total-contracts').textContent = totalContracts;
  const totalPnlEl = document.getElementById('closed-total-pnl');
  totalPnlEl.innerHTML = `<span class="${pnlClass(totalPnl)}">${(totalPnl >= 0 ? '+' : '') + fmt$(totalPnl)}</span>`;
  tfoot.style.display = '';
}
```
